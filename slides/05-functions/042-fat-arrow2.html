Fixing this

Let's fix this using ()=>. 


// TypeScript

var car = {
    all: ['Model S', 'Golf'],
    selected: '',
    select(carType: string) {
        this.all.forEach((t) =>  {
            if (t === carType) {
                this.selected = t;
            }
         });
    }
};

car.select('Model S')
car.selected); // => "Model S"
window['selected']); // => "undefined"


// JavaScript

var car = {
    all: ['Model S', 'Golf'],
    selected: '',
    select: function (carType) {
        var _this = this;
        this.all.forEach(function (t) {
            if (t === carType) {
                _this.selected = t;
            }
        });
    }
};

Rule of thumb: when in doubt, use arrow functions
